47 /* fribidi_join_arabic - do Arabic joining
48  *
49  * This function does the Arabic joining algorithm. Means, given Arabic
50  * joining types of the characters in ar_props (don't worry,
51  * FriBidiJoiningType can be casted to FriBidiArabicProp automagically), this
52  * function modifies this properties to grasp the effect of neighboring
53  * characters. You probably need this information later to do Arabic shaping.
54  *
55  * This function implements rules R1 to R7 inclusive (all rules) of the Arabic
56  * Cursive Joining algorithm of the Unicode standard as available at
57  * http://www.unicode.org/versions/Unicode4.0.0/ch08.pdf#G7462. It also
58  * interacts correctly with the bidirection algorithm as defined in Section
59  * 3.5 Shaping of the Unicode Bidirectional Algorithm available at
60  * http://www.unicode.org/reports/tr9/#Shaping.
61  *
62  * There are a few macros defined in fribidi-joining-types.h for querying the
63  * Arabic properties computed by this function.
64  */
65 FRIBIDI_ENTRY void fribidi_join_arabic(
66  const FriBidiCharType *bidi_types, /* input list of bidi types as
67  returned by
68  fribidi_get_bidi_types() */
69  const FriBidiStrIndex len, /* input string length */
70  const FriBidiLevel *embedding_levels, /* input list of embedding
71  levels, as returned by
72  fribidi_get_par_embedding_levels */
73  FriBidiArabicProp *ar_props /* Arabic properties to analyze, initilized by
74  joining types, as returned by
75  fribidi_get_joining_types */
76  );
