136 #define fribidi_remove_bidi_marks FRIBIDI_NAMESPACE(remove_bidi_marks)
137 /* fribidi_remove_bidi_marks - remove bidi marks out of an string
138  *
139  * This function removes the bidi and boundary-neutral marks out of an string
140  * and the accompanying lists. It implements rule X9 of the Unicode
141  * Bidirectional Algorithm available at
142  * http://www.unicode.org/reports/tr9/#X9, with the exception that it removes
143  * U+200E LEFT-TO-RIGHT MARK and U+200F RIGHT-TO-LEFT MARK too.
144  *
145  * If any of the input lists are NULL, the list is skipped. If str is the
146  * visual string, then positions_to_this is positions_L_to_V and
147  * position_from_this_list is positions_V_to_L; if str is the logical
148  * string, the other way. Moreover, the position maps should be filled with
149  * valid entries.
150  *
151  * A position map pointing to a removed character is filled with \-1. By the
152  * way, you should not use embedding_levels if str is visual string.
153  *
154  * For best results this function should be run on a whole paragraph, not
155  * lines; but feel free to do otherwise if you know what you are doing.
156  * Deprecated. Use fribidi_remove_special_chars instead.
157  *
158  * Returns: New length of the string, or \-1 if an error occured (memory
159  * allocation failure most probably).
160  */
161 FRIBIDI_ENTRY FriBidiStrIndex fribidi_remove_bidi_marks(
162  FriBidiChar *str, /* input string to clean */
163  const FriBidiStrIndex len, /* input string length */
164  FriBidiStrIndex *positions_to_this, /* list mapping positions to the
165  order used in str */
166  FriBidiStrIndex *position_from_this_list, /* list mapping positions from the
167  order used in str */
168  FriBidiLevel *embedding_levels /* list of embedding levels */
169  ) FRIBIDI_GNUC_WARN_UNUSED FRIBIDI_GNUC_DEPRECATED;
